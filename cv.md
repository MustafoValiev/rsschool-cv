    function digitalRoot(n) { 
        if (n.toString().length === 1) {
            return n
        } else {
            answer = 0;
            for (let i = 0; i < n.toString().length; i++) {
                answer += +(n.toString()[i])
            }
            return digitalRoot(answer)
        }
    }
